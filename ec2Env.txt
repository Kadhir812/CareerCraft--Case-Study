name: CareerCrafter CI/CD

on:
  push:
    branches:
      - master
  pull_request:
    branches:
      - master
  workflow_dispatch:

permissions:
  contents: read
  id-token: write

env:
  AWS_REGION: ${{ vars.AWS_REGION }}
  EC2_HOST: ${{ vars.EC2_HOST }}
  EC2_USER: ${{ vars.EC2_USER }}
  EC2_SSH_KEY: ${{ secrets.EC2_SSH_KEY }}
  DOCKERHUB_USERNAME: ${{ vars.DOCKERHUB_USERNAME }}
  DOCKERHUB_BACKEND_REPOSITORY: ${{ vars.DOCKERHUB_BACKEND_REPOSITORY }}
  DOCKERHUB_FRONTEND_REPOSITORY: ${{ vars.DOCKERHUB_FRONTEND_REPOSITORY }}

jobs:
  test-and-scan:
    runs-on: ubuntu-latest
    services:
      mysql:
        image: mysql:8.0
        env:
          MYSQL_DATABASE: career
          MYSQL_ROOT_PASSWORD: root
        ports:
          - 3306:3306
        options: >-
          --health-cmd="mysqladmin ping -h 127.0.0.1 -uroot -proot"
          --health-interval=10s
          --health-timeout=5s
          --health-retries=10

    steps:
      - name: Checkout code
        uses: actions/checkout@v4
        with:
          fetch-depth: 0

      - name: Set up Java 21
        uses: actions/setup-java@v4
        with:
          distribution: temurin
          java-version: 21
          cache: maven

      - name: Set up Node.js 20
        uses: actions/setup-node@v4
        with:
          node-version: 20
          cache: npm
          cache-dependency-path: frontend_careercrafter/package-lock.json

      - name: Install frontend dependencies
        working-directory: frontend_careercrafter
        run: npm ci --legacy-peer-deps

      - name: Build frontend
        working-directory: frontend_careercrafter
        env:
          VITE_API_BASE_URL: /api/v1
        run: npm run build

      - name: Run backend tests and generate coverage
        working-directory: backend_careercrafter
        env:
          DB_URL: jdbc:mysql://127.0.0.1:3306/career
          DB_USERNAME: root
          DB_PASSWORD: root
          JWT_SECRET: ci-test-secret-key-ci-test-secret-key-ci-test-secret-key
          JWT_EXPIRATION_MS: 86400000
          AWS_REGION: ap-south-1
          S3_BUCKET: ci-test-bucket
        run: bash ./mvnw -B clean verify

      - name: Run SonarQube scan
        if: github.event_name == 'push'
        continue-on-error: true
        working-directory: backend_careercrafter
        env:
          SONAR_TOKEN: ${{ secrets.SONAR_TOKEN }}
          SONAR_HOST_URL: ${{ vars.SONAR_HOST_URL }}
          DB_URL: jdbc:mysql://127.0.0.1:3306/career
          DB_USERNAME: root
          DB_PASSWORD: root
          JWT_SECRET: ci-test-secret-key-ci-test-secret-key-ci-test-secret-key
          JWT_EXPIRATION_MS: 86400000
          AWS_REGION: ap-south-1
          S3_BUCKET: ci-test-bucket
        run: |
          bash ./mvnw -B org.sonarsource.scanner.maven:sonar-maven-plugin:4.0.0.4121:sonar \
            -Dsonar.host.url=${SONAR_HOST_URL} \
            -Dsonar.token=${SONAR_TOKEN} \
            -Dsonar.projectKey=${{ vars.SONAR_PROJECT_KEY }} \
            -Dsonar.organization=${{ vars.SONAR_ORGANIZATION }} \
            -Dsonar.coverage.jacoco.xmlReportPaths=target/site/jacoco/jacoco.xml

  build-and-push-images:
    needs: test-and-scan
    if: github.event_name == 'push'
    runs-on: ubuntu-latest

    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Log in to Docker Hub
        uses: docker/login-action@v3
        with:
          username: ${{ secrets.DOCKERHUB_USERNAME }}
          password: ${{ secrets.DOCKERHUB_TOKEN }}

      - name: Build and push backend image
        uses: docker/build-push-action@v6
        with:
          context: ./backend_careercrafter
          file: ./backend_careercrafter/Dockerfile
          push: true
          tags: |
            ${{ env.DOCKERHUB_USERNAME }}/${{ env.DOCKERHUB_BACKEND_REPOSITORY }}:${{ github.sha }}
            ${{ env.DOCKERHUB_USERNAME }}/${{ env.DOCKERHUB_BACKEND_REPOSITORY }}:latest

      - name: Build and push frontend image
        uses: docker/build-push-action@v6
        with:
          context: ./frontend_careercrafter
          file: ./frontend_careercrafter/Dockerfile
          push: true
          build-args: |
            VITE_API_BASE_URL=/api/v1
          tags: |
            ${{ env.DOCKERHUB_USERNAME }}/${{ env.DOCKERHUB_FRONTEND_REPOSITORY }}:${{ github.sha }}
            ${{ env.DOCKERHUB_USERNAME }}/${{ env.DOCKERHUB_FRONTEND_REPOSITORY }}:latest

  deploy-to-ec2:
    needs: build-and-push-images
    if: github.event_name == 'push'
    runs-on: ubuntu-latest

    steps:
      - name: Deploy to EC2 via SSH
        uses: appleboy/ssh-action@v1.0.3
        env:
          DOCKERHUB_USERNAME: ${{ env.DOCKERHUB_USERNAME }}
          DOCKERHUB_BACKEND_REPOSITORY: ${{ env.DOCKERHUB_BACKEND_REPOSITORY }}
          DOCKERHUB_FRONTEND_REPOSITORY: ${{ env.DOCKERHUB_FRONTEND_REPOSITORY }}
          DOCKERHUB_TOKEN: ${{ secrets.DOCKERHUB_TOKEN }}
        with:
          host: ${{ env.EC2_HOST }}
          username: ${{ env.EC2_USER }}
          key: ${{ env.EC2_SSH_KEY }}
          envs: DOCKERHUB_USERNAME,DOCKERHUB_BACKEND_REPOSITORY,DOCKERHUB_FRONTEND_REPOSITORY,DOCKERHUB_TOKEN
          script_stop: true
          script: |
            set -e

            echo "$DOCKERHUB_TOKEN" | docker login --username "$DOCKERHUB_USERNAME" --password-stdin

            cd ~/Backend
            sudo -n docker compose down --remove-orphans || true
            sudo -n docker compose pull
            sudo -n docker compose up -d --remove-orphans
